			% Syntaxe d'utilisation
			\synt{du}{-sh}{}
			
			% Description de la commande
			\desc{\begin{itemize}
					\item Affiche l'espace mémoire utilisé par un fichier ou un répertoire.
					\item L'option \lin{-h} (human readable) convertie l'affichage des tailles mémoires en unités conventionnelles (en nombre de blocs par défaut).
					\item L'option \lin{-s} (sumurize) n'affiche pas le détail des fichiers et des sous-répertoires.
				\end{itemize}
				}
			
			% Exemple d'uitlisation
			\expl{
				\begin{center}
					\mprompt{
						\prompt{du -sh Documents/}{5,2G	Documents/}
						\prompt{\cursor}{}
					}
				\end{center}
			}
