			% Syntaxe d'utilisation
			\synt{env}{}{}
			
			% Description de la commande
			\desc{\begin{itemize}
					\item Affiche la liste des variables d'environnement et leurs valeurs.
				\end{itemize}
				}
			
			% Exemple d'utilisation
			\expl{
				\begin{center}
					\mprompt{
						\prompt{env}{TERM=xterm\\SHELL=/bin/bas\\HOME=/home/chez\_moi\\LANGUAGE=fr\_FR.UTF-8\\PATH=/usr/local/bin:/usr/bin:/bin:/opt/bin:/home/chez\_moi/bin}
						\prompt{\cursor}{}
					}
				\end{center}
			}
