			% Syntaxe d'utilisation
			\synt{tar}{xv}{nom\_archive }
			
			% Description de la commande
			\desc{\begin{itemize}
					\item Extrait les fichiers et répertoires d'une archive.
					\item Les fichiers sont placés dans le répertoire courant.
					\item L'option \lin{x} (e\textbf{X}tarct) indique que la commande \lin{tar} doit utiliser un algorithme de désarchivage.
				\end{itemize}
				}
			
			% Exemple d'utilisation
			\expl{
				Extrait le contenu de l'archive \lin{espace.tar} :\\\vspace{5pt}
				\begin{center}
					\mprompt{
						\prompt{tar xv espace.tar}{}
					}
				\end{center}
			}
